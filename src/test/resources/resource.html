Example: Javascript »

A simple ajax (JSONP) request to the data API using jQuery.

  var data = {
    resource_id: 'ad05d144-74ec-4a56-9391-574c5295a111', // the resource id
    limit: 5, // get 5 results
    q: 'jones' // query for 'jones'
  };
  $.ajax({
    url: 'http://dati.trentino.it/api/action/datastore_search',
    data: data,
    dataType: 'jsonp',
    success: function(data) {
      alert('Total results found: ' + data.result.total)
    }
  });
